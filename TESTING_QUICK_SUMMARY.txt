  --watch PATH            # Watch for changes

  --verbose               # Verbose output

7. TEST PATTERNS
================================================================================

Unit Test Pattern:
  module Unit.Parse.ExpressionTest (tests) where
  
  tests :: TestTree
  tests = testGroup "Parse.Expression"
    [ testCase "int literal" $ parseExpr "42" @?= Right (Int 42)
    , testCase "float literal" $ parseExpr "3.14" @?= Right (Float 3.14)
    ]

Property Test Pattern:
  module Property.Canopy.VersionProps (tests) where
  
  tests :: TestTree
  tests = testGroup "Version Properties"
    [ testProperty "roundtrip" $ \v -> 
        Version.fromChars (Version.toChars v) == Just v
    ]

Integration Test Pattern:
  - Tests multiple components together
  - File system operations
  - Project initialization
  - Build integration

Golden Test Pattern:
  - Compare output with expected files
  - Input sources in test/Golden/sources/
  - Expected output in test/Golden/expected/
